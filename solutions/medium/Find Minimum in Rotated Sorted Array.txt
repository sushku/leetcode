// Title: Find Minimum in Rotated Sorted Array
            // Difficulty: Medium
            // Language: Python3
            // Link: https://leetcode.com/problems/find-minimum-in-rotated-sorted-array/

class Solution:
    def findMin(self, nums: List[int]) -> int:
        prevNum, first = nums[0], nums[0]
        for i in range(1, len(nums)):
            if nums[i] < prevNum:
                return nums[i]
            prevNum = nums[i]
        return first
