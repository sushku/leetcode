// Title: Two Sum
            // Difficulty: Easy
            // Language: Python3
            // Link: https://leetcode.com/problems/two-sum/

class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        d = {}
        for i in range(len(nums)):
            diff = target - nums[i]
                return [d[diff], i]
            else:
            if diff in d:
                d[nums[i]] = i
